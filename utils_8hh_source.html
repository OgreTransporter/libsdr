890 class GWNSource: public Source
891 {
892 public:
894  GWNSource(double sample_rate, size_t buffer_size=1024)
895  : Source(), _sample_rate(sample_rate), _buffer_size(buffer_size), _buffer(_buffer_size),
896  _mean(0)
897  {
898  // Seed RNG
899  std::srand(std::time(0));
900  // Determine mean
901  switch (Config::typeId<Scalar>()) {
902  case Config::Type_u8:
903  case Config::Type_cu8:
904  case Config::Type_u16:
905  case Config::Type_cu16:
906  _mean = 1;
907  break;
908  default:
909  _mean = 0;
910  break;
911  }
912  // Propergate config
913  setConfig(Config(Config::typeId<Scalar>(), _sample_rate, _buffer_size, 1));
914  }
915 
917  virtual ~GWNSource() {
918  _buffer.unref();
919  }
920 
924  void next() {
925  double a,b;
926  for (size_t i=0; i<_buffer_size; i+=2) {
927  _getNormal(a,b);
928  _buffer[i] = Scalar(Traits<Scalar>::scale*(a+_mean));
929  _buffer[i+1] = Scalar(Traits<Scalar>::scale*(b+_mean));
930  }
931  if (_buffer_size%2) {
932  _getNormal(a,b); _buffer[_buffer_size-1] = Scalar(Traits<Scalar>::scale*(a+_mean));
933  }
934  this->send(_buffer, true);
935  }
936 
937 
938 protected:
940  void _getNormal(double &a, double &b) {
941  // Obtain pair of std. normal floating point values
942  double x = 2*(double(std::rand())/RAND_MAX) - 1;
943  double y = 2*(double(std::rand())/RAND_MAX) - 1;
944  double s = std::sqrt(x*x + y*y);
945  while (s >= 1) {
946  x = 2*(double(std::rand())/RAND_MAX) - 1;
947  y = 2*(double(std::rand())/RAND_MAX) - 1;
948  s = std::sqrt(x*x + y*y);
949  }
950  a = x*std::sqrt(-2*log(s)/s);
951  b = y*std::sqrt(-2*log(s)/s);
952  }
953

960  Buffer<Scalar> _buffer;
962  double _mean;
